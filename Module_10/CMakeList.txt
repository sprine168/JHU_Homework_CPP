cmake_minimum_required(VERSION 3.10)
project(Module10_Elevator)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED True)

cmake_policy(SET CMP0167 NEW)

# Search for Boost.Log and Boost.System
find_package(Boost 1.70 REQUIRED COMPONENTS log system)

if(Boost_FOUND)
    message(STATUS "Boost found! Include directories: ${Boost_INCLUDE_DIRS}")
else()
    message(FATAL_ERROR "Boost not found.")
endif()

add_executable(Module10_Elevator main.cpp)

# Link headers and compiled libraries to your executable
target_include_directories(Module10_Elevator PRIVATE ${Boost_INCLUDE_DIRS})
target_link_libraries(Module10_Elevator PRIVATE ${Boost_LIBRARIES})
